#include "Stack.hpp"
using std::string;
using std::cout;
using std::endl;


template <typename E>
Stack<E>::Stack() {
    list=nullptr;
    num_elements=0;
    first=nullptr;
    last=nullptr;
}

template <typename E>
Stack<E>::~Stack() {
    while(first!=nullptr){
        Node *temp=first;
        first=temp->next;
        delete temp;
    }
}

template <typename E>
bool Stack<E>::is_empty() const {
    return first== nullptr;
}

template <typename E>
E Stack<E>::top() {
    if(is_empty()){
        throw StackException("Top","Stack is empty Bozo");
    }
    return last->element;
}

template <typename E>
void Stack<E>::push(E & item) {
    Node *ptr = new Node(item, nullptr);
    if(is_empty()){
        first=ptr;
        last=ptr;
    }
    else {
        first->next=ptr;
        first = ptr;
    }
} template <typename E>
E Stack<E>::pop() {
    if(is_empty()){
        throw StackException("pop","Stack is empty Bozo");
    }
    Node *temp=first; //creates a temporary pointer for fun
    // (definitely for fun and not because it's actually needed)

    first=temp->next; //makes first point to the next node in the list
    E q=temp->element; //retrieves the element from the node to be deleted
    delete temp; //deletes the node
    if(first!=nullptr) {
        first->next = nullptr; //setting first back to null
    }
    if(first==nullptr){
        last= nullptr;
    }
    return q; //returns the element
}

template class Stack<string> ;
template class Stack<int> ;
