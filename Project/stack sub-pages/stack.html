#ifndef STACK_H
#define STACK_H

#include <iostream>
using std::string;
using std::ostream;

class StackException {
    private:
    string location;
    string msg;

    public:
    StackException(const string& whereThrown, const string & message) {
        location = whereThrown;
        msg = message;
    }
    string to_string() const {
    return "*** StackException in "+
    location + ": "+msg;
}
    friend ostream& operator<< (ostream & out, const StackException & se) {
        out << se.to_string();
        return out;
    }
};


template <typename E>
class Stack {
        private:
        // for the array version only
        static const int INIT_CAPACITY = 3; // array version only
        int capacity;
        E * list;

    // for the linked list version only
    class Node {
        public:

        E element;
        Node* next;

    Node(const E & element, Node * next) {
        this->element = element;
        this->next = next;
    }
};
    Node * first;
    Node * last;
    // for either version
    int num_elements; // either version
    public:
        Stack();

        // in this assignment, no copies of the stack are allowed.
Stack(const Stack & copy) = delete;

~Stack();

bool is_empty() const;

E top();

void push(E & element);

E pop();

};

#endif
