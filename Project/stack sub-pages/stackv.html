// The Array version of the ADT Stack with all elements push(), pop(), is_empty()

template <typename E>
Stack<E>::Stack() {
    capacity=INIT_CAPACITY;
    num_elements=0;
    list=new E[capacity];
} //creating all the required variables and things for the stack

template <typename E>
Stack<E>::~Stack() {
    delete[] list;
} //destroys the list

template <typename E>
bool Stack<E>::is_empty() const { //This will check if it is empty
    return num_elements==0;//Returns true if empty and false if not
}

template <typename E>
E Stack<E>::top() {
    if(is_empty()){
        throw StackException("Top","Stack is empty Bozo");
    } //Throws exception if Stack is empty since we cannot look at something in an empty box
    return list[num_elements-1];
}//shows the top element of the stack

template <typename E>
void Stack<E>::push(E & item) {
    if(num_elements==capacity){
        capacity=capacity*2; //doubles the size of the array if full because we like using a lot of space
        E* templist=new E[capacity];
        for(int i=0;i             templist[i]=list[i];
        }
    delete list;
    list=templist;
    }
    list[num_elements++]=item; //adding the element to the array because thats kinda the point idk
}

template <typename E>
E Stack<E>::pop() {
    if(is_empty()){
        throw StackException("pop","Stack is empty Bozo");
    } //WHY IS THE USER TRYING TO REMOVE SOMETHING FROM AN EMPTY ARRAY
    num_elements--;
    return list[++num_elements];
}//removes the top element from the stack


template class Stack<string> ;
template class Stack<int> ;
